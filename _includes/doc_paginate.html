{%- comment -%} Previous / next links for docs. Reads the same sidebar data the left nav uses (`site.data.sidebars[page.sidebar]`), flattens the nested subitems into one ordered list of URLs, finds the current page, and only renders a side (previous or next) if that neighbour actually exists. If neither exists the whole block is omitted. {%- endcomment -%} {%- assign sidebar = site.data.sidebars[page.sidebar] -%} {%- if sidebar -%} {%- assign flat_urls = "" | split: "" -%} {%- for item in sidebar.subitems -%} {%- if item.url -%} {%- assign flat_urls = flat_urls | push: item.url -%} {%- endif -%} {%- if item.subitems -%} {%- for sub in item.subitems -%} {%- if sub.url -%} {%- assign flat_urls = flat_urls | push: sub.url -%} {%- endif -%} {%- endfor -%} {%- endif -%} {%- endfor -%} {%- assign current_index = -1 -%} {%- for u in flat_urls -%} {%- if u == page.url -%} {%- assign current_index = forloop.index0 -%} {%- break -%} {%- endif -%} {%- endfor -%} {%- if current_index >= 0 -%} {%- assign prev_index = current_index | minus: 1 -%} {%- assign next_index = current_index | plus: 1 -%} {%- assign total = flat_urls | size -%} {%- assign has_prev = false -%} {%- if current_index > 0 -%} {%- assign prev_url = flat_urls[prev_index] -%} {%- assign prev_doc = site.docs | where: "url", prev_url | first -%} {%- if prev_doc -%}{%- assign has_prev = true -%}{%- endif -%} {%- endif -%} {%- assign has_next = false -%} {%- if next_index < total -%} {%- assign next_url = flat_urls[next_index] -%} {%- assign next_doc = site.docs | where: "url", next_url | first -%} {%- if next_doc -%}{%- assign has_next = true -%}{%- endif -%} {%- endif -%} {%- if has_prev or has_next -%}
{%- if has_prev -%}
{{ prev_doc.title }}
← {{ site.data.translation[site.lang].previous | default: "Previous" }}
{%- endif -%} {%- if has_next -%}
{{ next_doc.title }}
{{ site.data.translation[site.lang].next | default: "Next" }} →
{%- endif -%}
{%- endif -%} {%- endif -%} {%- endif -%}
